% Header
\newcommand{\header}[5]{
  \begin{tabular*}{\textwidth}{l@{\extracolsep{\fill}}r}
    \textbf{\href{#1}{\Large #2}} & \textbf{Email:} \href{mailto:#3}{#3} \\
    \href{#1}{#1} & \textbf{Mobile:} \href{tel:#4}{#4} \\
  \end{tabular*}
  \vspace{0.5em}
  \textit{\small #5}
  \vspace{0.5em}
}

% Subheading
\newcommand{\subheading}[4]{
  \item
    \begin{tabular*}{0.97\textwidth}[t]{l@{\extracolsep{\fill}}r}
      \textbf{#1} & #2 \\
      \textit{\small#3} & \textit{\small #4} \\
    \end{tabular*}\vspace{-5pt}
}

% Entry
\newcommand{\entry}[2]{
  \item\small{
    \textbf{#1}{: #2 \vspace{-2pt}}
  }
}
